void transform(Matrix3 t) {
  final Vector2 center = new Vector2.zero();
  final Vector2 halfExtents = new Vector2.zero();
  copyCenterAndHalfExtents(center, halfExtents);
  t
    ..transform2(center)
    ..absoluteRotate2(halfExtents);
  _min
    ..setFrom(center)
    ..sub(halfExtents);
  _max
    ..setFrom(center)
    ..add(halfExtents);
}
