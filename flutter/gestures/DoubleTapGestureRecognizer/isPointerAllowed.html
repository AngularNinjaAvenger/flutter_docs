bool isPointerAllowed (PointerEvent event)
override

Checks whether or not a pointer is allowed to be tracked by this recognizer.

Implementation

@override
bool isPointerAllowed(PointerEvent event) {
  if (_firstTap == null) {
    switch (event.buttons) {
      case kPrimaryButton:
        if (onDoubleTap == null)
          return false;
        break;
      default:
        return false;
    }
  }
  return super.isPointerAllowed(event);
}
